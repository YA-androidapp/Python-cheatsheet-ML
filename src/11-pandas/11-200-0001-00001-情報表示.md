import numpy as np
import pandas as pd

arr = np.arange(50).reshape(5, 10)
idx = pd.Index(['r{}'.format(x+1) for x in range(5)], name = 'index')
col = pd.Index(['c{}'.format(x+1) for x in range(10)], name= 'column')
df = pd.DataFrame(arr, index=idx, columns=col) # 2次元: データフレーム

df.info()

# 統計量
df.describe()
